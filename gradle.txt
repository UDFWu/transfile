  duplicatesStrategy 'INCLUDE'

  // 專案列表
  def projectList = ['scsb.ebank.database', 'scsb.ebank.eai', 'scsb.ebank.common', 'scsb.ebank.front.main']
  projectList.each {
    projectName -> {
      from(project(":${projectName}").file('src/main/resources')){
        include ("**/*")
      }
      into "$buildDir/libs/config"
    }
  }
}
 
jar{
  enabled = false
}

allprojects {

  	apply plugin: 'java'
  	apply plugin: 'java-library'
  	apply plugin: 'war'
  	apply plugin: 'maven-publish'

	group = 'scsb.ebank.front'
	version = '1.0.0'
	sourceCompatibility = '17'
	targetCompatibility = '17'
	ext['spring-security.version']='5.8.11'
	compileJava.options.encoding = 'UTF-8'
	compileTestJava.options.encoding = 'UTF-8'
	
    repositories {

				name = "nexus"

			}
    	} else if (project.hasProperty("cicd") && project.cicd == "pro") {
	        maven {
				name = "nexus"
				url = uri("https://sdpsn01.cbss.scsb.com.tw/repository/maven-public")
			}
    	} else {
	        mavenCentral()
    	}    	
    }

    dependencies {
    	
        // librarys
        implementation fileTree(dir: '../../libs', include: ['*.jar'])
        
        implementation platform('scsb.ebank:ebank-ap-bom:1.0')
    	
        // spring boot web base
        implementation 'org.springframework.boot:spring-boot-starter'
        implementation ('org.springframework.boot:spring-boot-starter-web'){
          exclude group: 'org.springframework', module: 'spring-web'
        }
        implementation 'org.springframework.boot:spring-boot-starter-validation'
        implementation ('org.springframework.boot:spring-boot-starter-security'){
          exclude group: 'org.springframework', module: 'spring-web'
        }
        implementation 'org.springframework.boot:spring-boot-starter-data-jpa'
        implementation 'org.springframework.boot:spring-boot-starter-mustache'
        implementation 'org.springframework.boot:spring-boot-starter-data-redis'

        // apache poi tool (Excel)

  	task clearDist() {
	  delete "$buildDir/libs"
	}
	
	def envList = []
	
	// creating build tasks by runtime environment profiles, e.g. application-dev.yml or application-sit.yml
	fileTree("src/main/resources") {
	  include "application-*.yml"
	}.each {
	  file ->
	    def propertyName = file.name
	  String envName = propertyName.substring(12, propertyName.lastIndexOf(".yml"))
	  envList.add('-' + envName)
	  task "build_${envName}"(dependsOn: ['build']) {
	    group = 'build'
	    description = "Assembles and tests this project for ${envName}"
	  }
	  task "bootRun_${envName}"(dependsOn: ['bootRun']) {
	    group = 'application'
	    description = "Runs this project as a Spring Boot application for ${envName}"
	  }
	}
	
	test {
	  useJUnitPlatform()
	}
	
	// reslove environment by build task name e.g. dev/test/prod
	ext.getRuntimeEnvironment = {
	  ->
	  try {
	    def envName = ''
	    def taskName = getTaskName().split('_')
	    if (taskName == null ||
	      (taskName?.length == 1 && taskName?.getAt(0) == 'bootRun') ||
	      (taskName?.length == 1 && taskName?.getAt(0) == 'build')) {
	      println '[WARN] Set envName = dev, first task > ' + project.gradle.startParameter.taskNames
	      envName = 'dev'
	    } else {
	      println '>> RuntimeEnv: ' + taskName
	      println '>> RuntimeEnv.len: ' + taskName?.length
	      envName = taskName?.last()
	      println '>> RuntimeEnv.last: ' + envName
	    }
	    return envName
	  } catch (ignored) {
	    return null;
	  }
	}
	
	// reslove first execute task name
	ext.getTaskName = {
	  ->
	  return project.gradle.startParameter.taskNames[0] ? project.gradle.startParameter.taskNames[0] : ''
	}

	  // 建立MANIFEST

	               'Built-On': new Date()

	jar {

	publishing {

	            version = '${version}'
	            pom.withXml {
				    def root = asNode()
				    def packagingNode = root.get('packaging')
				
				    if (packagingNode.isEmpty()) {
				        // 沒有 packaging 就加
				        root.appendNode('packaging', 'war')
				    } else {
				        // 有就修改現有的值
				        packagingNode[0].value = 'war'
				    }
				}
	        }
	    }
	}
	
	tasks.named('build') {
    	dependsOn 'generatePomFileForMavenPomOnlyPublication'	
	}
