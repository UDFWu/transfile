                                 returnStdout: true
                         ).trim()

                             mvn -B -q -U clean install \\
                             -pl :${project}-${env.TYPE} \\

        stage('Dependency Check') {
            when {
                expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
            }

            steps {
                // 執行 Dependency Check
                dependencyCheck additionalArguments: '--data /opt/dependency-check/data -f HTML JSON --noupdate --disableNodeAudit --disableCentral --disableOssIndex', nvdCredentialsId: 'NVD-API-KEY', odcInstallation: 'owasp_check'
            }
        }

         stage('Build APP Image') {
             when {
                 expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
             }

             steps {
                 script {
                     withCredentials([usernamePassword(credentialsId: 'harbor-user', passwordVariable: 'HARBOR_PASSWORD', usernameVariable: 'HARBOR_USER')]) {
                         // 先登入 Harbor，爲了拉取 base image
                         sh "podman login ${CONTAINER_REGISTRY} -u '${HARBOR_USER}' -p '${HARBOR_PASSWORD}'"
                         for (project in deployProjects) {
                             echo "Building ${project}-${env.TYPE} image"

                             sh """
                                 podman images | grep ${project}-${env.TYPE} | awk '{print \$3}' | xargs podman rmi --force || true
                                 cd ${env.WORKSPACE}/${project}/${project}-${env.TYPE}
                                 podman build --build-arg CONTAINER_REGISTRY=${CONTAINER_REGISTRY} -f ./Dockerfile -t ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG} .
                             """
                         }
                     }
                 }
             }
         }

         stage('Release APP Image') {
             when {
                 expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
             }

             steps {
                 script {
                     // 上傳 image 到 Harbor
                     for (project in deployProjects) {
                         echo "Releaseing ${project}-${env.TYPE} image"

                         //取得 RepoDigest
                         def repoDigest = sh(returnStdout: true, script: "podman inspect -f '{{index .RepoDigests 0}}' ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG}").trim()
                         // 上傳 repoDigest 到 Harbor
                         sh "podman push ${repoDigest}"
                         // 上傳 image 到 Harbor
                         sh "podman push ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG}"
                         withCredentials([file(credentialsId: 'cosign-private-key', variable: 'COSIGN_PRIVATE_KEY'), string(credentialsId: 'cosign-password', variable: 'COSIGN_PASSWORD')]) {
                             println(repoDigest)
                             // 使用 cosign 簽名
                             sh "cosign sign --key '${COSIGN_PRIVATE_KEY}' --tlog-upload=false --allow-insecure-registry=true ${repoDigest}"
                         }
                     }
                 }
             }
         }

        stage('Trivy scan APP image') {
            when {
                expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
            }

            steps {
                script {
                    for (project in deployProjects) {
                        sh 'mkdir -p reports'
                        // 使用 Trivy 進行掃描 image
                        // --skip-db-update: 跳過更新 Trivy 的資料庫
                        // --skip-java-db-update: 跳過更新 Java 的資料庫
                        // --offline-scan: 離線掃描
                        // --skip-check-update: 跳過檢查更新
                        // --format template: 輸出格式為 template
                        // --template '@/usr/local/share/trivy/templates/html.tpl': 使用 HTML 的格式
                        // template 要有讀的權限
                        // -o reports/trivy-image-report.html: 輸出檔案為 reports/trivy-image-report.html
                        reportFile = "reports/${env.HARBOR_PROJECT}-${project}-${env.TYPE}-report.html"
                        sh "trivy image --skip-db-update \
                            --skip-java-db-update \
                            --offline-scan \
                            --skip-check-update \
                            --format template \
                            --template '@/usr/local/share/trivy/templates/html.tpl' \
                            -o ${reportFile} \
                            ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG}"
                        // 上傳報告到 Jenkins
                        publishHTML([
                                allowMissing: true,
                                alwaysLinkToLastBuild: false,
                                keepAll: true,
                                reportDir: 'reports',
                                reportFiles: "${project}-${env.TYPE}-report.html",
                                reportName: "${project}-${env.TYPE} Report",
                                reportTitles: "${project}-${env.TYPE} Report",
                                useWrapperFileDirectly: true
                        ])
                    }

                    sh "podman logout ${CONTAINER_REGISTRY}"
                }
            }
        }

        stage('Clone OCP YAML Repository') {
            steps {
                script {
                    sh "rm -rf ./*"
                }
                checkout scmGit(
                        branches: [[name: '*/sit']],
                        browser: github("${OCP_GIT_REPO_URL}"),
                        extensions: [],
                        userRemoteConfigs: [[credentialsId: 'gitlab-user', url: "${OCP_GIT_REPO_URL}"]]
                )
            }
        }

        stage('Update the OCP YAML') {
            steps {
                script {
                    for (project in deployProjects) {
                        echo "Processing YAML for project: ${project}-${TYPE}"
                        // 替換 YAML 文件中的佔位符
                        def workspacePath = "${env.WORKSPACE}/${project}/${project}-${TYPE}/${env.PROFILE}"
                        def imageHash = sh(returnStdout: true, script: "podman inspect -f '{{index .RepoDigests 0}}' ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG}").trim()
                        sh """
                            cd ${workspacePath}
                            sed -i 's|{{IMAGE_VERSION}}|${imageHash}|g' deployment.yaml
                        """
                        echo "Updated image: ${imageHash}"
                    }
                }
            }
        }

        stage('Publish to Nexus') {
            steps {
                script {
                    for (project in deployProjects) {
                        sh """
                            cp base/${PROFILE}/configmap.yaml ${project}/${project}-${TYPE}/${PROFILE}/base-configmap.yaml
                            cp base/${PROFILE}/secret.yaml ${project}/${project}-${TYPE}/${PROFILE}/base-secret.yaml
                            zip -r -9 ${project}-${TYPE}.zip ${project}/${project}-${TYPE}/${PROFILE}/
                            mvn deploy:deploy-file "-DgroupId=com.ibm.ncbs" \
                                "-DartifactId=${project}-${TYPE}" \
                                "-Dversion=${env.TAG}" \
                                "-Dpackaging=zip" \
                                "-Dfile=./${project}-${TYPE}.zip" \
                                "-Durl=${NEXUS_URL}/repository/maven-releases/" \
                                "-Dmaven.wagon.http.ssl.insecure=true" \
                                "-Dmaven.wagon.http.ssl.allowall=true" \
                                "-DrepositoryId=serverid" \
                                "-DgeneratePom=true"
                        """
                    }
                }
            }
        }

        stage('Get Lastest Project Info') {
            steps {
                script {
                    APP_VERSION=env.TAG
                    url = sh(script: "echo ${OCP_GIT_REPO_URL} | sed 's,https://,,'", returnStdout: true).trim()
                    group_id = sh(script: "echo ${url} | grep / | cut -d/ -f2", returnStdout: true).trim()
                    project_id = sh(script: "echo ${url} | grep / | cut -d/ -f3 | sed 's,.git,,'", returnStdout: true).trim()

                    response = httpRequest httpMode: 'GET', contentType: 'APPLICATION_JSON',
                            url: "${CICDAPISERVICE_URL}/api/v1/packages/projects?group=${group_id}&project=${project_id}&limit=1",
                            validResponseCodes: '200:500'
                    println('Response: ' + response.content)
                    if (response.status != 200) {
                        error(response)
                    }

                    jsonObj = readJSON text: response.content

                    if (!jsonObj.items) {
                        jsonObj = [
                                items: [[
                                                group    : group_id,
                                                project  : project_id
                                        ]],
                                meta: [limit: 1, page: 1, total: 1]
                        ]
                    }

                    //Jenkins pipeline要新增 def projectInfo
                    projectInfo = jsonObj.items[0]
                    echo "${projectInfo}"
                }
            }
        }

        stage('Build Change Project Info') {
            steps {
                script {
                    def changes = []
                    deployProjects.each {project ->
                        dir ("${project}/${project}-${TYPE}") {
                            def projectName = "${project}-${TYPE}"
                            def nexusUrl = "${NEXUS_URL}/repository/maven-releases/com/ibm/ncbs/${project}-${TYPE}/${env.TAG}/${project}-${TYPE}-${env.TAG}.zip"
                            def imageHash = sh(returnStdout: true, script: "podman inspect -f '{{index .RepoDigests 0}}' ${CONTAINER_REGISTRY}/${env.HARBOR_PROJECT}/${project}-${env.TYPE}:${env.TAG}").trim()
                            changes << [name: projectName, url: nexusUrl, hash: imageHash]
                        }
                    }
                    echo "${changes}"

                    def item = projectInfo
                    if (item.packages == null) item.packages = []

                    changes.each { c ->
                        def idx = item.packages.findIndexOf { it.name == c.name }
                        if (idx == -1) {
                            item.packages << [name: (c.name as String), url: (c.url as String), hash: (c.hash as String)]
                        } else {
                            item.packages[idx].url  = (c.url as String)
                            item.packages[idx].hash  = (c.hash as String)
                        }
                    }

                    item.version = (env.TAG ?: '0.0.0') as String
                    item.packages = item.packages.collect { p ->
                        [
                            name: (p.name as String),
                            url : (p.url  as String),
                            hash : (p.hash  as String)
                        ]
                    }

                    def jsonBody = writeJSON(returnText: true, json: item, pretty: 1)
                    env.jsonPayload = jsonBody
                    echo "${jsonBody}"
                }
            }
        }

         stage('Rollout Deployment') {
             when {
                 expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
             }

             steps {
                 script {
                     openshift.withCluster(OPENSHIFT_SERVER) {
                         openshift.withProject("${env.NAMESPACE}") {
                             def rolloutSuccess = true
                             def meta = readJSON(text: env.jsonPayload)
                             def pkgMap = [:]
                             meta.packages.each { p ->
                                 pkgMap[(p.name as String)] = [
                                     url : (p.url  as String)
                                 ]
                             }

                             for (project in deployProjects) {
                                String svc = "${project}-${TYPE}"
                                def pkg = pkgMap[svc]

                                sh """
                                    curl -k ${pkg.url} --output ${project}-${TYPE}.zip
                                    unzip -o ${project}-${TYPE}.zip -d ${project}-${TYPE}
                                """
                                
                                def workspacePath = "${env.WORKSPACE}/${project}-${TYPE}/${project}/${project}-${TYPE}/${env.PROFILE}"
                                echo "${workspacePath}"
                                try {
                                    // openshift.verbose(true)
                                    openshift.raw('apply', '-f', "${workspacePath}/base-configmap.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/base-secret.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/configmap.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/secret.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/deployment.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/service.yaml")
                                    openshift.raw('apply', '-f', "${workspacePath}/servicemonitor.yaml")
                                    openshift.raw('rollout', 'restart', "deployment/${env.HARBOR_PROJECT}-${project}-${env.TYPE}")
                                    echo "Rollout of ${env.HARBOR_PROJECT}-${project}-${TYPE} succeeded!"
                                } catch (Exception e) {
                                    echo "Rollout of ${env.HARBOR_PROJECT}-${project}-${TYPE} failed: ${e.getMessage()}"
                                    rolloutSuccess = false // 設爲失敗
                                }
                             }

                             // 確認所有 Rollout 狀態
                             if (rolloutSuccess) {
                                 echo 'Rollout successful!'
                             } else {
                                 error('One or more Rollout checks failed.')
                             }
                         }
                     }
                 }
             }
         }

         stage('Check Deployment and Pod Status') {
             when {
                 expression { env.TYPE == 'app' || env.TYPE == 'rpt' }
             }

             steps {
                 script {
                     openshift.withCluster(OPENSHIFT_SERVER) {
                         openshift.withProject("${env.NAMESPACE}") {
                             for (project in deployProjects) {
                                 echo "Check ${project} deployment and Pod status"

                                 try {
                                     def deployment = openshift.selector('deployment', "${env.HARBOR_PROJECT}-${project}-${env.TYPE}")
                                     def deploymentObject = openshift.selector('deployment', "${env.HARBOR_PROJECT}-${project}-${env.TYPE}").object()
                                     // 10分鐘 timeout
                                     def rolloutStatus = timeout(10) {deployment.rollout().status()}

                                     // 判斷是否成功
                                     if (rolloutStatus.out.contains('successfully rolled out')) {
                                         echo "Deployment ${env.HARBOR_PROJECT}-${project}-${env.TYPE} has successfully rolled out."
                                         echo "rolloutStatus: ${rolloutStatus.out}"
                                     } else {
                                         // 提供更詳細的錯誤資訊
                                         echo "Rollout of ${env.HARBOR_PROJECT}-${project}-${env.TYPE} failed. Status: ${deploymentObject}"
                                         error("Rollout of ${env.HARBOR_PROJECT}-${project}-${env.TYPE} failed. Status: ${deployment}")
                                     }
                                 } catch (Exception e) {
                                     echo "Deployment or Pod check for ${env.HARBOR_PROJECT}-${project}-${env.TYPE} failed: ${e.getMessage()}"
                                     error("Rollout of ${env.HARBOR_PROJECT}-${project}-${env.TYPE} failed. Status: ${deployment}")
                                     break // 停止檢查，直接退出迴圈
                                 } finally {
                                     // openshift.verbose(false) // 關閉 verbose 模式
                                 }
                             }
                             echo 'All deployments and Pods are successful!'
                         }
                     }
                 }
             }
         }

        stage('Save project info') {
            steps {
                script {
                    response = httpRequest acceptType: 'APPLICATION_JSON', contentType: 'APPLICATION_JSON',
                            httpMode: 'POST',
                            requestBody: "${env.jsonPayload}",
                            url: "${CICDAPISERVICE_URL}/api/v1/packages/project",
                            validResponseCodes: '200:500'
                    println('Response: '+response.content)
                    if (response.status != 200) {
                        error(response)
                    }
                    echo "${env.jsonPayload}"
                }
            }
        }
    }

    post {
        success {
            script {
                sh """
                    cd ..
                    rm -r -f swagger
                    mkdir -p swagger
                    cd swagger
                """

                dir('../swagger') {
                    checkout scmGit(
                            branches: [[name: '*/main']],
                            browser: github("${GIT_SWAGGER_REPO_URL}"),
                            extensions: [],
                            userRemoteConfigs: [[credentialsId: 'gitlab-user', url: "${GIT_SWAGGER_REPO_SSH_URL}"]]
                    )

                    def timestamp = sh(script: 'date +%Y%m%d%H%M%S', returnStdout: true).trim()
                    sh "mkdir -p ${timestamp}"

                    for (project in deployProjects) {
                        try {
                            sh """
                                cp ${env.WORKSPACE}/${project}/${project}-${TYPE}/target/openapi/${project}-openapi.json ${timestamp}/
                            """
                        } catch (err) {
                            echo "Failed to read openapi.json for ${project}: ${err}"
                        }
                    }

                    withCredentials([gitUsernamePassword(credentialsId: 'mid-write', gitToolName: 'Default')]) {
                        sh '''
                            git checkout main
                            git add .
                            git commit -m 'add new swagger' || echo "Nothing to commit"
                            git push origin main
                        '''
                    }
                }

                echo "${projectVersions}"
            }
        }
        always {
            // 歸檔報告
            archiveArtifacts artifacts: '**/dependency-check-report.html, **/checkmarx-report.pdf', allowEmptyArchive: true
        }
    }
}
