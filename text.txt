stage('Publish All Artifacts to Nexus') {
    steps {
        script {
            // 遍歷所有專案目錄
            for (project in deployProjects) {
                // 自動判定倉庫
                // 註：這裡假設所有模組版本一致，若不一致，可改用 mvn help:evaluate 抓取
                def targetRepo = env.TAG.contains("SNAPSHOT") ? "maven-snapshots" : "maven-releases"

                echo "正在搜尋並上傳 ${project} 模組的產出物..."

                // 使用 find 尋找該專案目錄下所有 target 裡的 jar/war
                // 排除掉不必要的 sources 或 javadoc
                def files = sh(
                    script: "find ./${project} -name 'target' -type d -exec find {} -maxdepth 1 -type f \\( -name '*.jar' -o -name '*.war' \\) ! -name '*-sources.jar' ! -name '*-javadoc.jar' \\; ",
                    returnStdout: true
                ).trim().split('\n')

                for (filePath in files) {
                    if (!filePath) continue
                    
                    def fileName = filePath.split('/').last()
                    // 從檔名推斷 artifactId (例如 cbr-app-0.0.1.jar -> artifactId 就是 cbr-app)
                    // 或者你也可以統一用 ${project}
                    def artifactId = fileName.replaceAll("-${env.TAG}.*", "")
                    def ext = fileName.endsWith('.war') ? 'war' : 'jar'

                    sh """
                        mvn deploy:deploy-file -DgroupId=mid.ai \
                            -DartifactId=${artifactId} \
                            -Dversion=${env.TAG} \
                            -Dpackaging=${ext} \
                            -Dfile=${filePath} \
                            -Durl=${NEXUS_URL}/repository/${targetRepo}/ \
                            -DrepositoryId=serverid \
                            -DgeneratePom=true \
                            -Dmaven.wagon.http.ssl.insecure=true
                    """
                }
            }
        }
    }
}
